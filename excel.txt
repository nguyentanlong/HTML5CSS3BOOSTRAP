# ====== Hàm tạo tên file mới ======
def get_new_filename(base_name="ketqua.xlsx"):
    if not os.path.exists(base_name):
        return base_name
    i = 1
    while True:
        new_name = f"ketqua-{i}.xlsx"
        if not os.path.exists(new_name):
            return new_name
        i += 1

def parse_input(input_text):
    products = [p.strip() for p in input_text.split(".") if p.strip()]
    parsed = []
    for prod in products:
        fields = [f.strip() for f in prod.split(",")]
        if len(fields) == 8:  # STT, Tên, Đvt, Mã hiệu, Xuất xứ, Bảo hành, SL, Đơn giá
            parsed.append(fields)
    return parsed

def upload_to_drive(file_path):
    gauth = GoogleAuth()
    gauth.LocalWebserverAuth()  # Mở trình duyệt để đăng nhập Google
    drive = GoogleDrive(gauth)

    file = drive.CreateFile({'title': os.path.basename(file_path)})
    file.SetContentFile(file_path)
    file.Upload()
    print(f"Đã upload {file_path} lên Google Drive với ID: {file['id']}")
    return file['id']
